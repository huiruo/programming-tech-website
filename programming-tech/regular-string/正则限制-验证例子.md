/*
* ## 1.正则匹配html
* */
let richText = `< div > test < p > <b> </b></p ></div > `
let reg = /<([a-z]+?)(?:\s+?[^>]*?)?>\s*?<\/\1>/ig;
while (reg.test(richText)) {
    richText = richText.replace(reg, "");
}
console.log("1.正则匹配html", richText)
